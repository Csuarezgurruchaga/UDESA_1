    e.last_name
    ,e.first_name
    ,d.department_name
    ,j.description
    ,e2.last_name
from employee e
inner join department d
    on e.department_number = d.department_number
inner join job j
    on e.job_code = j.job_code
inner join employee e2
    on e.manager_employee_number = e2.employee_number
